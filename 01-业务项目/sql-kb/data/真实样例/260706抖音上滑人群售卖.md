    iceberg_zjyprc_hadoop.tmp.newhome_douyin_thirdapp_open_2026_0706
select
    t0.did,
    t0.oaid,
    t0.oaid sale_id
from
    (
        select
            s2.distinct_id as did,
            s2.oaid
        from s1
        join s2 
        on s1.did=s2.distinct_id ----取oaid
        join s3 
        on s1.did = s3.distinct_id
    ) as t0
    left join t1 on t0.oaid=t1.oaid
    left join t2 on t0.did=t2.did
    left join t3 on t0.did=t3.did
    left join t4 on t0.did=t4.did
    left join t5 on t0.did=t5.did
    left join t6 on t0.did=t6.did
where
    t1.oaid is null ----剔除已售卖人群
    and t2.did is null ----剔除一线/新一线城市用户
    and t3.did is null ----剔除两年内设备用户
    and t4.did is null ----剔除月vv>1000高活用户
    and t5.did is null ----剔除月收入>1.16元用户
    and t6.did is null ----剔除高端机用户
group by
    t0.did,
    t0.oaid,
    t0.oaid
