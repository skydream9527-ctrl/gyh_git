-- 按渠道 + 新老用户拆分次日留存
SELECT channel, user_type, version,
  COUNT(DISTINCT uid) AS base,
  SUM(is_d1_retained) / COUNT(DISTINCT uid) AS d1_rate
FROM dwd_user_retention
WHERE version IN ('v2.2','v2.3')
GROUP BY 1,2,3
